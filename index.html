function jouer(joueur) { let ordi; // choix aléatoire let hasard = Math.floor(Math.random() * 3); if (hasard === 0) { ordi = "pierre"; } else if (hasard === 1) { ordi = "feuille"; } else { ordi = "ciseaux"; } // résultat if (joueur === ordi) { document.getElementById("resultat").innerText = "Égalité"; } else if (joueur === "pierre" && ordi === "ciseaux") { document.getElementById("resultat").innerText = "Gagné"; } else if (joueur === "feuille" && ordi === "pierre") { document.getElementById("resultat").innerText = "Gagné"; } else if (joueur === "ciseaux" && ordi === "feuille") { document.getElementById("resultat").innerText = "Gagné"; } else { document.getElementById("resultat").innerText = "Perdu"; } }
